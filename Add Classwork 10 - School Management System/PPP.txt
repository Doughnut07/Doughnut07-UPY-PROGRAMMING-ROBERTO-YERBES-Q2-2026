#PROCESS
logged       ← FALSE
current_user ← ""

WHILE logged IS EQUAL TO FALSE DO
    GET user     FROM user
    GET password FROM user

    IF user IS IN users THEN
        IF users[user]['password'] IS EQUAL TO password THEN
            logged       ← TRUE
            current_user ← user
        ELSE
            DISPLAY "\nIncorrect password. Try again.\n"
        ENDIF
    ELSE
        DISPLAY "\nWrong User/Password. Try again.\n"
    ENDIF
ENDWHILE

user_role ← users[current_user]['rol']

#OUTPUT
DISPLAY "\nWelcome, " + users[current_user]['name'] + " (" + user_role + ")"

IF user_role IS EQUAL TO 'student' THEN
    DISPLAY "\nReport card for " + users[current_user]['name']
    aprobadas ← EMPTY SET
    pendientes ← EMPTY SET

    FOR EACH subject_name IN subjects DO
        grade ← notes[current_user][subject_name]
        DISPLAY subject_name + ": " + grade

        IF grade >= 7.0 THEN
            ADD subject_name TO aprobadas
        ELSE
            ADD subject_name TO pendientes
        ENDIF
    ENDFOR

    DISPLAY "\nPassed subjects: " + aprobadas
    DISPLAY "Pending subjects: " + pendientes

ELSE IF user_role IS EQUAL TO 'professor' THEN
    DISPLAY "\nStudent list:"
    FOR EACH student_key IN users DO

    GET student_to_grade FROM user

            DISPLAY "- " + subject_name
        ENDFOR

        GET subject_to_grade FROM user

        IF subject_to_grade IS EQUAL TO 'exit' THEN
            DISPLAY "\nExiting program."
        ELSE IF subject_to_grade IS IN subjects THEN
            GET new_grade_input FROM user

            IF new_grade_input IS EQUAL TO 'exit' THEN
                DISPLAY "\nExiting program."
            ELSE
                new_grade    ← CONVERT new_grade_input TO decimal
                GET confirmation FROM user

                IF confirmation IS EQUAL TO 'exit' THEN
                    DISPLAY "\nExiting program."
                ELSE IF confirmation IS EQUAL TO 'yes' THEN
                    notes[student_to_grade][subject_to_grade] ← new_grade
                    DISPLAY "\nGrade updated."
                ELSE IF confirmation IS EQUAL TO 'no' THEN
                    DISPLAY "\nModification cancelled."
                ELSE
                    DISPLAY "\nInvalid option. Modification cancelled."
                ENDIF
            ENDIF
        ELSE
            DISPLAY "\nSubject not found."
        ENDIF
    ELSE

    ENDIF
